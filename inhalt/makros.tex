{
\ifdefined\wortlang 
\else
\newlength{\wortlang}
\fi
\settowidth{\wortlang}{#1}
\ifdim \wortlang<0.5\linewidth
\begin{tabular}{p{\wortlang}p{\linewidth-4\tabcolsep-\wortlang}}
\else
\begin{tabular}{p{0.5\linewidth-2\tabcolsep}p{0.5\linewidth-2\tabcolsep}}
\fi
#1 & #2
\end{tabular}
\let\wortlang\relax
}
